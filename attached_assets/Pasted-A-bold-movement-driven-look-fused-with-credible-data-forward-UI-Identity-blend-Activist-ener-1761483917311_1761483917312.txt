Score Badge
// src/components/ScoreBadge.tsx
interface ScoreBadgeProps { score: number | null; size?: "sm"|"md"|"lg"; }


export function ScoreBadge({ score, size="md" }: ScoreBadgeProps) {
  const bucket = score == null ? "none" : score >= 70 ? "high" : score >= 40 ? "mid" : "low";
  const map = {
    none: { ring: "ring-slate-300", text: "text-slate-500", label: "No data" },
    high: { ring: "ring-emerald-400", text: "text-emerald-600", label: "Strong" },
    mid:  { ring: "ring-amber-400", text: "text-amber-600", label: "Mixed" },
    low:  { ring: "ring-rose-400", text: "text-rose-600", label: "Risk" },
  }[bucket];


  const dim = size === "lg" ? "w-16 h-16 text-lg" : size === "sm" ? "w-9 h-9 text-xs" : "w-12 h-12 text-sm";


  return (
    <div className={`grid place-items-center rounded-full ring-4 ${map.ring} bg-white ${dim}`}>
      <div className={`font-heading font-semibold ${map.text}`}>{score ?? "‚Äì"}</div>
      <div className="sr-only">Impact score: {score ?? "No data"} ({map.label})</div>
    </div>
  );
}
